ACME con Caddy

caddy:
  image: caddy:latest
  container_name: crux-caddy
  volumes:
    - ./caddy/Caddyfile:/etc/caddy/Caddyfile
    - caddy_data:/data
    - caddy_config:/config
    - cert_data:/certs
  ports:
    - "80:80"
    - "443:443"

# Caddyfile
mail.example.com {
  reverse_proxy nginx:80
  tls admin@local.com
}

CA Interna

# Generar RA de confianza interna
openssl genrsa -out ca.key 4096
openssl req -new -x509 -days 365 -key ca.key -out ca.crt \
  -subj "/C=US/ST=Local/O=Crux Webmail/CN=Crux CA"

# Generar cert para cada servicio
openssl genrsa -out service.key 2048
openssl req -new -key service.key -out service.csr \
  -subj "/C=US/ST=Local/O=Crux Webmail/CN=crux-server"
openssl x509 -req -days 365 -in service.csr \
  -CA ca.crt -CAkey ca.key -set_serial 01 \
  -out service.crt

Key Rotation

# Cron para rotación automática
0 0 1 * * /usr/local/bin/rotate-certificates.sh

# rotate-certificates.sh
#!/bin/bash
ACME_RENEW=0
CADDY_CERT=$(date -r /certs/live/mail.example.com/cert.pem +%s)
if [ $(( $(date +%s) - $CADDY_CERT )) -gt 21600000 ]; then
  ACME_RENEW=1
fi

if [ $ACME_RENEW = 1 ]; then
  caddy renew --force
  docker compose restart nginx
fi
