Reglas Estrictas Activas

FlagQué previene
strictNullChecksAcceder a propiedades de null/undefined
noUncheckedIndexedAccessIndexado sin check: arr[0] puede ser undefined
exactOptionalPropertyTypesAsignar undefined a propiedades opcionales
noImplicitOverrideOverride accidental en métodos heredados
verbatimModuleSyntaximport type obligatorio para tipos
noUncheckedIndexedAccessIndex signatures con posible undefined
isolatedModulesCódigo no transpilable por tsc solo

Prácticas de Tipado

// ✅ Buenas prácticas
import type { User, Session } from "@types/auth";
import type { FastifyRequest } from "fastify";

// Tipos explícitos en interfaces
interface MailMessage {
  id: string;
  from: string;
  to: string[];
  subject: string;
  body: string;
  attachments: Array<{
    filename: string;
    data: Buffer;
    contentType: string;
  }>;
  createdAt: Date;
  flags: string[];
}

// Generics para respuestas API
interface ApiResponse {
  data: T;
  meta?: {
    count: number;
    page: number;
  };
  error?: {
    code: string;
    message: string;
  };
}

// Type guards
function isEmailError(err: unknown): err is EmailError {
  return err instanceof Error && "emailCode" in err;
}

// Utility types
export type Unpacked = T extends (infer U)[] ? U : T;
export type DeepPartial = { [K in keyof T]?: T[K] extends object ? DeepPartial : T[K] };

Anti-Patterns Prohibidos

// ❌ Prohibido
const foo: any = getData();
const bar = (arg: Object) => {
  console.log(arg.value);
};
type Bad = { prop: string | undefined; }; // preferir { prop?: string; }

// ✅ Corregido
const foo: ReturnType = getData();
const bar = (arg: { value: string }) => {
  console.log(arg.value);
};
type Good = { prop?: string; };

Type-Checking en CI

# Package.json scripts
{
  "scripts": {
    "typecheck": "tsc --noEmit",
    "typecheck:verbose": "tsc --noEmit --listFiles"
  }
}
# En CI: npm run typecheck
